let data = {}; function calculatePoints(runs, sr){ let points = runs; if(sr < 100) points -= 2; else if(sr >= 300) points += 5; else if(sr >= 200) points += 3; else if(sr >= 100) points += 1; return points; } function addPlayer(){ let name = document.getElementById("player").value; let runs = parseInt(document.getElementById("runs").value); let sr = parseFloat(document.getElementById("sr").value); if(!runs || !sr){ alert("Enter runs and strike rate"); return; } let matchPoints = calculatePoints(runs, sr); // ✅ aggregate if(data[name]){ data[name].matches += 1; data[name].runs += runs; data[name].totalSR += sr; data[name].points += matchPoints; // 🔥 sum points (NOT recalculated) } else { data[name] = { matches: 1, runs: runs, totalSR: sr, points: matchPoints }; } render(); } function render(){ let div = document.getElementById("mvpList"); div.innerHTML = ""; let arr = []; for(let name in data){ let p = data[name]; let avgSR = p.totalSR / p.matches; arr.push({ name, matches: p.matches, runs: p.runs, sr: avgSR.toFixed(1), pts: p.points // ✅ already summed }); } // sort by points arr.sort((a,b)=> b.pts - a.pts); arr.forEach((p,i)=>{ div.innerHTML += `
${i===0 ? "🏆 " : ""}${p.name}
Matches: ${p.matches} | Runs: ${p.runs}
Avg SR: ${p.sr}
Points: ${p.pts}
`; }); }
